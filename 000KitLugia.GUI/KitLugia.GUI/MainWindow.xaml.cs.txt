using System.Windows;
using System.Windows.Controls;
using KitLugia.GUI.Pages;

// Aliases para resolver conflitos entre WinForms e WPF
using Button = System.Windows.Controls.Button;
using RadioButton = System.Windows.Controls.RadioButton;
using Application = System.Windows.Application;

namespace KitLugia.GUI
{
    public partial class MainWindow : Window
    {
        public MainWindow()
        {
            InitializeComponent();

            // Navega para o Dashboard inicial
            MainFrame.Navigate(new DashboardPage());

            // "Ouve" o evento de clique do botão OK dentro do controle LugiaMsgBox
            // Quando clicado, escondemos o Overlay
            InternalMsgBox.OkClicked += (s, e) => CloseOverlay();
        }

        // --- SISTEMA DE OVERLAY (AVISO INTEGRADO) ---

        // Método chamado por outras páginas (ex: ShowNotification("Erro tal"))
        public void ShowNotification(string message, string title = "AVISO DO SISTEMA")
        {
            InternalMsgBox.SetContent(message, title);
            OverlayContainer.Visibility = Visibility.Visible;

            // System.Media.SystemSounds.Exclamation.Play(); // <-- COMENTE OU APAGUE ESTA LINHA
        }

        private void CloseOverlay()
        {
            OverlayContainer.Visibility = Visibility.Collapsed;
        }

        // Fecha se clicar no fundo preto
        private void Overlay_MouseDown(object sender, System.Windows.Input.MouseButtonEventArgs e)
        {
            CloseOverlay();
        }

        // --- SISTEMA DE NAVEGAÇÃO ---

        // Evento de clique na Sidebar
        private void NavButton_Click(object sender, RoutedEventArgs e)
        {
            if (sender is RadioButton btn)
            {
                string tag = btn.Tag?.ToString() ?? "";
                NavigateToPage(tag);
            }
        }

        // Método Público que as Páginas chamam para mudar a tela e o botão
        public void NavigateToPage(string pageTag)
        {
            switch (pageTag)
            {
                case "🏠":
                    if (BtnDashboard.IsChecked != true) BtnDashboard.IsChecked = true;
                    MainFrame.Navigate(new DashboardPage());
                    break;

                case "⚡":
                    if (BtnTweaks.IsChecked != true) BtnTweaks.IsChecked = true;
                    MainFrame.Navigate(new TweaksPage());
                    break;

                case "📱":
                    if (BtnApps.IsChecked != true) BtnApps.IsChecked = true;
                    MainFrame.Navigate(new BloatwarePage());
                    break;

                case "💿":
                    if (BtnStorage.IsChecked != true) BtnStorage.IsChecked = true;
                    MainFrame.Navigate(new CleanupPage());
                    break;

                case "🌐":
                    if (BtnNetwork.IsChecked != true) BtnNetwork.IsChecked = true;
                    MainFrame.Navigate(new NetworkPage());
                    break;

                default:
                    // Placeholder para ferramentas que não existem
                    // Se tiver BtnTools, pode adicionar case "🛠️" aqui
                    ShowNotification("Página em desenvolvimento.", "AGUARDE");
                    break;
            }
        }

        // --- JANELA CUSTOMIZADA ---
        private void BtnClose_Click(object sender, RoutedEventArgs e) => Application.Current.Shutdown();
        private void BtnMaximize_Click(object sender, RoutedEventArgs e) => WindowState = (WindowState == WindowState.Normal) ? WindowState.Maximized : WindowState.Normal;
        private void BtnMinimize_Click(object sender, RoutedEventArgs e) => WindowState = WindowState.Minimized;
    }
}
